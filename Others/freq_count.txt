find the most occurence element in array
ex:[1,2,3,4,5,3,2,2,4,4,3,2,2,2,2,2,8]

let arr = [1,2,3,4,5,3,2,2,4,4,3,2,2,2,2,2,8];
let cnt = 0, count = 1;
let temp;
for(let i = 0; i < arr.length; i++){
	for(let j = i; j < arr.length; j++){
		if(arr[i] === arr[j])
			cnt++;
		if(count < cnt)
		{
			count = cnt;
			temp = arr[i];
		}
	}
	cnt = 0;
}
`${temp} ${count} times`;
